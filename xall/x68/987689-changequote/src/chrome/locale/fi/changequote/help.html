MUKAUTETUN PÄIVÄYSMUODON SYNTAKSI

%d = kuukaudenpäivä, 2 numeroa (”01” - ”31”)
%D = viikonpäivä tekstinä, englanninkielinen,
     3 kirjainta (”Fri”)
%m = kuukausi (”01” - ”12”)
%M = kuukausi tekstinä, englanninkielinen,
     3 kirjainta (”Jan”)
%y = vuosi, 2 numeroa (”99”)
%Y = vuosi, 4 numeroa (”1999”)
%i = minuutit (”00” - ”59”)
%s = sekunnit (”00” - ”59”)
%H = tunnit, 24 tunnin muoto (”00” - ”23”)
%h = tunnit, 12 tunnin muoto (”01” - ”12”)
%A = ”AM” tai ”PM”
%a = ”am” tai ”pm”
%e = space-filled date of month
%z = timezone offset from UTC
